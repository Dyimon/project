using System.Linq;
using System.Windows;
using System.Windows.Controls;
using WpfConstructor.Clases;
using System;
using System.Collections.Generic;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Data;
using System.Windows.Documents;
using System.Windows.Input;
using System.Windows.Media;
using System.Windows.Media.Imaging;
using System.Windows.Shapes;

namespace WpfConstructor
{
    /// <summary>
    /// Логика взаимодействия для WinTicket.xaml
    /// </summary>
    public partial class WinTicket : Window
    {
        int countRecord;//Количество записей в общес списке из базы данных
        int record = 4;//Количество записей в конечном списке
        int numberList = 1;//Выбранный номер страницы
        public WinTicket()
        {
            InitializeComponent();
            FillTable(numberList);//Передаётся номер начальной страницы
        }
        private void FillTable(int number)
        {
            var listCity = Helper.GetContext().City.ToList();
            //Начало функции постраничной навигации (после всех фильтров)
            countRecord = listCity.Count();
            if(countRecord > record)
            {
                go();
                ListNumber();
                if (countRecord - (number - 1) * record >= record) { DgCity.ItemsSource = listCity.GetRange(0 + (number - 1) * record, record); }
                else { DgCity.ItemsSource = listCity.GetRange(0 + (number - 1) * record, countRecord - (number - 1) * record); }
            }
            else { DgCity.ItemsSource = listCity;}
            //Конец функции постраничной навигации 
        }
        private void ListNumber()
        {
            BtnNumber1.FontWeight = FontWeights.Normal;
            BtnNumber2.FontWeight = FontWeights.Normal;
            BtnNumber3.FontWeight = FontWeights.Normal;
            BtnNumber4.FontWeight = FontWeights.Normal;
            BtnNumber5.FontWeight = FontWeights.Normal;
            if (int.Parse(BtnNumber1.Content.ToString()) == numberList) { BtnNumber1.FontWeight = FontWeights.Bold; }
            if (int.Parse(BtnNumber2.Content.ToString()) == numberList) { BtnNumber2.FontWeight = FontWeights.Bold; }
            if (int.Parse(BtnNumber3.Content.ToString()) == numberList) { BtnNumber3.FontWeight = FontWeights.Bold; }
            if (int.Parse(BtnNumber4.Content.ToString()) == numberList) { BtnNumber4.FontWeight = FontWeights.Bold; }
            if (int.Parse(BtnNumber5.Content.ToString()) == numberList) { BtnNumber5.FontWeight = FontWeights.Bold; }
        }
        private void go()
        {
            if(numberList >= 3 && numberList <= countRecord * 1.0 / record-1)
            {
                BtnNumber1.Content = numberList -2;
                BtnNumber2.Content = numberList -1;
                BtnNumber3.Content = numberList;
                BtnNumber4.Content = numberList +1;
                BtnNumber5.Content = numberList +2;
            }
        }
        private void Button_Click(object sender, RoutedEventArgs e)//Кнопка выхода
        {
            MainWindow mainWindow = new MainWindow();
            mainWindow.Show();
            this.Close();
        }
        private void BtnBack_Click(object sender, RoutedEventArgs e)//Кнопка назад
        {
            if (numberList > 1) { numberList--; FillTable(numberList); }
        }
        private void BtnNext_Click(object sender, RoutedEventArgs e)//Кнока вперед
        {
            double list = countRecord*1.0 / record;
            if (numberList < list) { numberList++; FillTable(numberList); }
        }
        private void BtnNumber_Click(object sender, RoutedEventArgs e)
        {
            numberList = int.Parse((sender as Button).Content.ToString());
            FillTable(numberList);
        }
    }
}
